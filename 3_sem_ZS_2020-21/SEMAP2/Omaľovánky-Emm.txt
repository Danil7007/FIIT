#include <stdio.h>

int minz2(int a, int b){
    return a>b ? b : a;
}
        
int main() {

    int x = 0;
    while (scanf("%d", &x)==1) {

        int c, z, m, minC=0, minZ=0, minM=0, minCdalsie, minZdalsie, minMdalsie;

        for (int i = 0; i < x; i++) {

            scanf("%d %d %d", &c, &z, &m);

            minCdalsie = minz2(minZ, minM);
            minZdalsie = minz2(minC, minM);
            minMdalsie = minz2(minC, minZ);

            minC = minCdalsie + c;
            minZ = minZdalsie + z;
            minM = minMdalsie + m;           
        }
        // najmenÅ¡ie z troch prvkov 
        if(minZ<minC)
            minC = minZ;
        if(minM<minC)
            minC = minM;
        printf("%d\n", minC); 
    }
    return 0;
}
